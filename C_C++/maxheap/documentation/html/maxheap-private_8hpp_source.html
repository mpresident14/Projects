164 template <typename T>
165 void MaxHeap<T>::bubbleDown(size_t index)
166 {
167  T*& parent = arr_[index];
168  T*& leftChild = arr_[2*index + 1];
169  T*& rightChild = arr_[2*index + 2];
170  size_t childIndex = 0;
171 
172  // If node is a leaf, can't go down any farther
173  if (2*index + 1 >= size_ && 2*index + 2 >= size_){
174  return;
175  }
176  // If node only has left child, check if they need to be swapped
177  else if (2*index + 2 >= size_){
178  if ( *parent < *leftChild ){
179  swapParentWithChild(index, true);
180  }
181  }
182  else{
183  // Figure out if left or right child is larger and get its index
184  bool left = true;
185  T*& largerChild = *leftChild < *rightChild ?
186  rightChild : leftChild;
187  if (largerChild == leftChild){
188  childIndex = 2*index + 1;
189  }
190  else{
191  childIndex = 2*index + 2;
192  left = false;
193  }
194  // Swap parent with larger child if either child is larger than parent
195  if ( *parent < *largerChild ){
196  swapParentWithChild(index, left);
197  bubbleDown(childIndex);
198  }
