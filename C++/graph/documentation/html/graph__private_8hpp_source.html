133 template <typename T, bool(*F)(const T&, const T&)>
134 forward_list<T> Graph<T,F>::getShortestPath(const T& start, const T& finish)
135 {
136  if ( getRelativeCount(start) == -1 || getRelativeCount(finish) == -1 ) {
137  throw std::invalid_argument("Params \"start\" and \"finish\" must be present in graph.");
138  }
139 
140  if ( getRelativeCount(start) == 0 || getRelativeCount(finish) == 0 ) {
141  return forward_list<T>();
142  }
143 
144  if (start == finish) {
145  return forward_list<T>( {start} );
146  }
147 
148  /*
149  * Dynamic approach: example
150  * a -> (b,c)
151  * b -> (a,c,f)
152  * c -> (a,b)
153  * d -> (c,e)
154  * e -> (d)
155  * f -> (b)
156  *
157  * Done when queue is empty.
158  * Map next item to previous item when we add to queue.
159  * Check if visited already before adding to the queue.
160  *
161  * a b c d e f
162  * a a a a ------->
163  * b b ------->
164  * c c ------->
165  * d d ------->
166  * e ------->
167  * f
168  *
169  * {a:nullptr, b:&a, c:&a, f:&b, d:&c, e:&d}
170  */
171 
172  // Just because we need start in the visited map (value doesn't matter)
173  unordered_map<T, T> visited( {{start, start}} );
174  queue<T> q;
175  q.push(start);
176 
177  while(!q.empty()) {
178  T& item = q.front();
179 
180  const unordered_set<T>* relatives = getRelatives(item);
181  for (const T& rel : *relatives) {
182 
183  // Traverse the visited map backwards to find path
184  if (rel == finish) {
185  forward_list<T> path( {finish} );
186  T&& prevNode = std::move(item);
187  while (prevNode != start) {
188  path.push_front(prevNode);
189  prevNode = visited.at(prevNode);
190  }
191 
192  path.push_front(start);
193  return path;
194  }
195 
196  // If has not been visited, map the relative to the item that led to it
197  if (visited.count(rel) == 0) {
198  visited.insert( {{rel, item}} );
199  q.push(rel);
200  }
201  }
202 
203  q.pop();
204  }
205 
206  // Return empty list if no path
207  return forward_list<T>();
208 }
